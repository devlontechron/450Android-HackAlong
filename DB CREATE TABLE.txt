DROP TABLE EVENTS;
CREATE TABLE EVENTS(
EID INTEGER(4) PRIMARY KEY, 
EName VARCHAR(500), 
ELink VARCHAR(500), 
ELocation VARCHAR(500),
ECreator VARCHAR(500), 
ETag VARCHAR(500), 
EDate VARCHAR(500), 
EDisc VARCHAR(500), 
EAttend INTEGER
);

DROP TABLE USER;
CREATE TABLE USER(
  UEmail VARCHAR(100) PRIMARY KEY,
  UPW VARCHAR(100),
  UName VARCHAR(100), 
  ULocation VARCHAR(100), 
  UAge INTEGER, 
  UEvents VARCHAR(100),
  UBio VARCHAR(500), 
  UTag VARCHAR(100)
);


DROP TABLE Tags;
CREATE TABLE Tags(
TID INTEGER PRIMARY KEY, 
TName VARCHAR(100)
);
